define(["require", "exports", "./CharacterModifier"], function (require, exports, CharacterModifier_1) {
    "use strict";
    /**
     * Created by Damien on 6/29/2016.
     */
    class Specialty extends CharacterModifier_1.CharacterModifier {
        constructor(name, characteristics, skills, talents, aptitudes, traits, kit, wounds, optionalModifiers) {
            super(characteristics, skills, talents, aptitudes, traits, kit, wounds, 0, CharacterModifier_1.OnlyWarCharacterModifierTypes.SPECIALTY);
            this._name = name;
            this._optionalModifiers = optionalModifiers;
        }
        apply(character) {
            super.apply(character);
            character.wounds.specialtyModifier = this.wounds;
            for (var entry of this.characteristics.entries()) {
                character.characteristics.get(entry[0]).specialtyModifier = entry[1];
            }
            ;
        }
        unapply() {
            this._appliedTo.wounds.specialtyModifier = 0;
            super.unapply();
        }
        ;
    }
    exports.Specialty = Specialty;
});
