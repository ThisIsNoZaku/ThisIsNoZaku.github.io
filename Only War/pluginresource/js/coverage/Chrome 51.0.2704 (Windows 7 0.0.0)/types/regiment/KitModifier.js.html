define(["require", "exports"], function (require, exports) {
    "use strict";
    /**
     * Created by Damien on 7/24/2016.
     */
    class KitModifier {
        constructor(description, kitPointCost, matcher, effect) {
            this._description = description;
            this._kitPointCost = kitPointCost;
            this._effect = effect;
            this._matcher = matcher;
        }
        get description() {
            return this._description;
        }
        get kitPointCost() {
            return this._kitPointCost;
        }
        canAffect(item) {
            return this._matcher(item);
        }
        apply(kit) {
            this._appliedTo = kit;
        }
        unapply() {
            this._appliedTo = null;
        }
    }
    exports.KitModifier = KitModifier;

    }
});
